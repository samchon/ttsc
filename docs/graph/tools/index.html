Tools

@ttsc/graph exposes four tools. Each is a narrowing engine over the checker-resolved graph: they return compact index records — symbol coordinates, relationships, and stable handles — not source bodies. Read source on demand with graph_expand. Each tool’s result is returned as JSON; the shapes below describe that JSON.

Every node and edge is resolved by the TypeScript checker — the graph is checker-resolved fact end to end, so there is no per-edge trust flag to weigh.

Pick the entry tool by what you know: graph_overview to orient, graph_query to find symbols from a natural query, graph_trace to follow flow, graph_expand to read the source the others located.

graph_overview

/**
 * A compact architecture map of the project — layers, hotspots, and the public
 * API surface (the exported types, named and ranked by how depended-on they
 * are) — with no source read. Call this first on an unfamiliar codebase: the
 * layers are the structure and the public API is the entry surface, so there is
 * no need to list files or open the entry module to orient.
 */
declare function graph_overview(props: OverviewProps): Overview;
 
interface OverviewProps {
  /** Facet to project, or `all` for every facet. @default "all" */
  aspect?: "all" | "layers" | "hotspots" | "publicApi";
}
 
interface Overview {
  /** Absolute project root. */
  project: string;
  /** Size of the graph. */
  counts: {
    files: number;
    nodes: number;
    edges: number;
    /** Node count per kind. */
    byKind: Record<string, number>;
  };
  /** Folder layering, largest first. */
  layers?: { dir: string; files: number; exported: number }[];
  /** Highest-dependency symbols, busiest first (structural edges excluded). */
  hotspots?: {
    id: string;
    name: string;
    kind: string;
    file: string;
    fanIn: number;
    fanOut: number;
  }[];
  /** Exported API symbols, most-depended-on first (test/typings/generated excluded). */
  publicApi?: { name: string; kind: string; file: string }[];
}

graph_query

/**
 * Find the symbols and clusters most relevant to a natural query, even without
 * the exact name. Mix code vocabulary and plain words; matches rank by
 * exact and dotted names, CamelCase/subword overlap, file path, and centrality.
 * Returns handles to follow with graph_expand or graph_trace.
 */
declare function graph_query(props: QueryProps): QueryResult;
 
interface QueryProps {
  /** What to find, e.g. `OrderService.create` or `repository find relations`. */
  query: string;
  /** Maximum hits to return. @default 12 */
  limit?: number;
}
 
interface QueryResult {
  hits: {
    /** Stable handle for graph_expand / graph_trace. */
    id: string;
    name: string;
    kind: string;
    file: string;
    /** 1-based declaration line, when known. */
    line?: number;
    /** Relative relevance; higher is a better match. */
    score: number;
  }[];
}

graph_trace

/**
 * Trace dependency flow from a symbol: `forward` to what it uses, `reverse` to
 * what uses it, or `impact` to the public API and tests a change would reach.
 * Follows real call/type edges only — structural edges are excluded. An
 * ambiguous start name returns its candidates instead of guessing.
 */
declare function graph_trace(props: TraceProps): TraceResult;
 
interface TraceProps {
  /** A node id from another tool, or a symbol name. */
  from: string;
  /** @default "forward" */
  direction?: "forward" | "reverse" | "impact";
  /** Hops to follow. @default 6 */
  maxDepth?: number;
  /** Cap on reached nodes. @default 60 */
  maxNodes?: number;
}
 
interface TraceResult {
  /** The resolved start node, or absent when `from` matched nothing. */
  start?: TraceNode;
  direction: string;
  /** Edges traversed, breadth-first. */
  hops: { from: string; to: string; kind: string; depth: number }[];
  /** Unique nodes reached, each with depth and impact roles. */
  reached: TraceNode[];
  /** True when the trace hit a bound and more flow exists. */
  truncated: boolean;
  /** When `from` was an ambiguous name, the matches to disambiguate with. */
  candidates?: TraceNode[];
}
 
interface TraceNode {
  id: string;
  name: string;
  kind: string;
  file: string;
  /** Hops from the start, on a reached node. */
  depth?: number;
  /** Why it matters to an impact trace: `exported`, `test`. */
  roles?: string[];
}

graph_expand

/**
 * Get the resolved shape of nodes another tool returned as handles: each one's
 * signature, and for a container its member outline. This is the graph's edge
 * over a file read — it returns structure, not inlined source. Pass every handle
 * in one call. The full body comes back only with `source: true`, for the few
 * methods whose logic you must read.
 */
declare function graph_expand(props: ExpandProps): ExpandResult;
 
interface ExpandProps {
  /** Node ids to expand, exactly as another tool returned them. */
  handles: string[];
  /** Also list each node's dependencies and dependents. @default false */
  neighbors?: boolean;
  /** Also return the full declaration source body. @default false */
  source?: boolean;
}
 
interface ExpandResult {
  nodes: {
    id: string;
    name: string;
    kind: string;
    file: string;
    /** 1-based declaration line. */
    line?: number;
    /** The declaration signature (head up to the body brace). */
    signature?: string;
    /** For a class/interface/namespace/enum/file: members, each with a signature. */
    members?: { name: string; kind: string; line?: number; signature?: string }[];
    /** Full declaration source — only when `source: true` was passed. */
    source?: string;
    /** True when `source` was cut at the line cap. */
    truncated?: boolean;
    /** Symbols this node uses. */
    dependsOn?: ExpandRef[];
    /** Symbols that use this node. */
    dependedOnBy?: ExpandRef[];
  }[];
  /** Handles that resolved to no node. */
  unknown: string[];
}
 
interface ExpandRef {
  id: string;
  name: string;
  kind: string;
  /** The edge kind connecting the two (`calls`, `type_ref`, …). */
  relation: string;
}

Node kinds

Symbol kinds are declarations the checker owns: file, package, namespace, module, function, class, interface, type, enum, variable, method, property, parameter. external_symbol is a dependency-boundary leaf. The graph does not interpret framework conventions; the checker-resolved decorator facts on a node are left for an agent to read.
